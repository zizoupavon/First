function submitQuiz() {

let score = 0;

const form = document.getElementById('quizForm');

const feedbackDiv = document.getElementById('feedback');

feedbackDiv.innerHTML = '';

const totalQuestions = Object.keys(correctAnswers).length;

 

for (let i = 1; i <= totalQuestions; i++) {

const qName = 'q' + i;

const selectedOption = form.elements[qName].value;

const correctOption = correctAnswers[qName];

const questionElement = document.querySelector(`input[name="${qName}"]:checked`).closest('.question-block');

// Remove previous styling

questionElement.classList.remove('correct', 'incorrect');

const resultItem = document.createElement('div');

resultItem.innerHTML = `<strong>Question ${i}:</strong> `;

 

if (selectedOption === correctOption) {

score++;

resultItem.classList.add('correct');

resultItem.innerHTML += `Correct!`;

} else {

resultItem.classList.add('incorrect');

resultItem.innerHTML += `Incorrect. You chose ${selectedOption}, but the correct answer is ${correctOption}.`;

}

const explanation = document.createElement('p');

explanation.classList.add('explanation');

explanation.textContent = 'Explanation: ' + explanations[qName];

resultItem.appendChild(explanation);

feedbackDiv.appendChild(resultItem);

}

 

document.getElementById('score').textContent = score;

document.getElementById('results').style.display = 'block';

// Scroll to results

document.getElementById('results').scrollIntoView({ behavior: 'smooth' });

}
